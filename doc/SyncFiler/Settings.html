def self.exist?(file="~/.syncfiler.d/settings.yml")
  f = File.expand_path file
  path = File.dirname(f)
  Dir.mkdir path unless Dir.exist? path
  File.exist? f
end

def self.write_setting_file( type, hash,
                             path='~/.syncfiler.d/settings.yml' )
  pt = File.expand_path(path)
  data = Hash.new
  fw = File.open(pt, 'w')
  yml = YAML.dump data
  fw.write yml
  fw.close
  return data
end
